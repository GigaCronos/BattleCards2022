HealthPoints=620;
Defense=267;
MinDamage=34;
MaxDamge=40;
Speed=170;
Evasion=25;

Atack(){
    int f=Random%(MaxDamge-MinDamage+1);
    return=MinDamage+f;
}
DealDamage(D){
    p=Random%100;
    if(p>Evasion){
        HealthPoints=HealthPoints-(D*100)/(100+Defense*5/100);
    }
    return=1;
    if(Health<=0){
        return=0;
    }
}
GetHealth(){
    return=Health;
}
GetSpeed(){
    return=Speed;
}
GetDamage(){
    return=(MinDamage+MaxDamge)/2;
}
GetDefense(){
    return=Defense;
}
